```
def get_sum(data):
    result = 0
    for datum in data:
       result = result + datum
    return result
   
def get_len(data):
    result = 0
    for datum in data:
        result = result + 1
    return result

def get_average(data):
    total = get_sum(data)
    n = get_len(data)
    result = total / n
    return result
    
score = [50, 60, 70]
average = get_average(score)
print(average)
```
